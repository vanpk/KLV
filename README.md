#clone src: 
git clone https://github.com/vanpk/KLV.git

#initialize git:
git init

#create branch:
git checkout -b <branch name>
  
#get source:
git pull

#to see status:
git status



#after changing file (add file):
git add <file name> 
#or add all file: 
git add --all 

#commit file:
git commit -m "<message>"

#push to branch:
git push origin <branch name>



#to know more on GIT, refer to https://github.com/joshnh/Git-Commands or google more.
