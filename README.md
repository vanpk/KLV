#clone src: 
git clone https://github.com/vanpk/KLV.git

#create branch:
git checkout -b <branch name>

#add file:
git add <file name> 
#or add all file: 
git add --all 

#commit file:
git commit -m "<message>"

#push to branch:
git push origin <branch name>

#to see status:
git status
